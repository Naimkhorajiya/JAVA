import java.util.Scanner;

class Fibonacci {
    void generateSeries(int n) {
        int a = 0, b = 1;
        System.out.print("Fibonacci Series: ");
        for (int i = 1; i <= n; i++) {
            System.out.print(a + " ");
            int c = a + b;
            a = b;
            b = c;
        }
    }
}

public class Task2 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter the number of terms in the Fibonacci series: ");
        int n = sc.nextInt();

        Fibonacci f = new Fibonacci();
        f.generateSeries(n);

        sc.close();
    }
}

class Factorial {
    int num;

    Factorial(int num) {
        this.num = num;
    }

    void calculateFactorial() {
        int fact = 1;
        for (int i = 1; i <= num; i++) {
            fact *= i;
        }
        System.out.println("The factorial of " + num + " is: " + fact);
    }
}

public class Task3 {
    public static void main(String[] args) {
        Factorial f = new Factorial(5);
        f.calculateFactorial();
    }
}

import java.util.Scanner;

class Vehicle {
    void display() {
        System.out.println("Vehicles are used for transportation.");
    }
}

class Car extends Vehicle {
    void display() {
        System.out.println("Cars are a type of vehicle.");
    }
}

public class Task7 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter '1' to create a Car object: ");
        int choice = sc.nextInt();

        if (choice == 1) {
            Car c = new Car();
            c.display();
        }

        sc.close();
    }
}
